```javascript
var
let
const
*/
//var : 언제든 재선언, 재할당이 가능하다.
var a; //선언
console.log(a) // undefined
a = 1; //할당
console.log(a);

// let : 재선언은 불가하고, 재할당은 가능함
let a; //선언
console.log(a) // undefined
let = 3; //할당
console.log(a);
```
